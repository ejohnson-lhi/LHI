# Diarization batch transcription dependencies.
#
# WhisperX bundles faster-whisper (CTranslate2-optimized Whisper) and
# pyannote diarization in one package. PyTorch is the heavy install
# (~2 GB) — pinned to the CPU-only build to keep the droplet's disk
# usage and Memory footprint reasonable. If a GPU is added later,
# swap to torch+cu*.
#
# Pinning versions because pyannote model + library compatibility is
# brittle across major versions; tested working set below.
#
# SYSTEM DEPENDENCIES (Debian/Ubuntu):
#   sudo apt install -y pkg-config ffmpeg \
#       libavformat-dev libavcodec-dev libavdevice-dev \
#       libavutil-dev libswresample-dev libswscale-dev libavfilter-dev
#
# These are needed because the av==11.* version transitively pinned by
# whisperx 3.1.5 → faster-whisper 1.0.1 has no Python 3.12 prebuilt
# wheel on PyPI, so pip falls back to building PyAV from source which
# needs pkg-config + ffmpeg dev headers. ffmpeg itself is also required
# at runtime by whisperx.load_audio() for transcoding non-WAV inputs.

# NOTE: whisperx was removed from this list — its 3.1.5 release was
# yanked (unofficial third-party upload) and has a hardcoded VAD model
# URL that's been moved and returns HTTP 301. We use the underlying
# components directly instead: faster-whisper (CTranslate2-optimized
# Whisper, listed below) for transcription + pyannote.audio for
# diarization + speaker embedding. faster-whisper has its own VAD
# (vad_filter=True). Less wrapper code, fewer broken URLs.

# pyannote.audio 4.0+ works with our diarize_batch.py since we call
# Pipeline.from_pretrained() directly. The 4.x embedder is WeSpeaker
# 512-dim (vs 3.x ECAPA-TDNN 192-dim). Cosine similarity normalizes
# internally so either works, BUT enrolled profiles must use the same
# embedder version as the batch run — re-enroll if you upgrade across
# the 3-to-4 boundary.
pyannote.audio>=3.3.0

# Torch CPU-only (let pip pick the latest compatible torch within whisperx's
# constraints; we tell pip to prefer the CPU index for the wheel).
--extra-index-url https://download.pytorch.org/whl/cpu
torch
torchaudio

# numpy for fingerprint dot products and saving .npy profiles.
numpy

# faster-whisper transitively, but pinned for explicitness.
faster-whisper

# Audio I/O — soundfile for embedding inference reads.
soundfile

# Used by enroll_speaker.py for clean audio loading + resampling.
librosa
